/*
자릿수 더하기
문제 설명
자연수 N이 주어지면, N의 각 자릿수의 합을 구해서 return 하는 solution 함수를 만들어 주세요.
예를들어 N = 123이면 1 + 2 + 3 = 6을 return 하면 됩니다.

제한사항
N의 범위 : 100,000,000 이하의 자연수
*/

// C++
#include <iostream>

using namespace std;
int solution(int n)
{
    int answer = 0;
    
    // 자릿수 문자열로 처리.
    string temp = to_string(n);
    for(int i = 0; i < temp.length(); i++){
        answer += temp[i]-48;
    }
    // 자릿수 나눗셈 연산으로 처리.
    answer = 0;
    while(n){
        answer += n%10;
        n /= 10;
    }
    
    // [실행] 버튼을 누르면 출력 값을 볼 수 있습니다.
    cout << "Hello Cpp" << endl;

    return answer;
}
